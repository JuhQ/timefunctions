### time functions


Returns different timespans in milliseconds.



#### Examples

```
var time = require('timefunctions');
```


```
time.seconds(60);
// 60000
```

```
time.minutes(60);
// 3600000
```

```
time.hours(24);
// 86400000
```

```
time.days(30);
// 2592000000
```

```
time.weeks(4);
// 2419200000
```

```
time.months(12);
// 31104000000
```

```
time.years(30);
// 946080000000
```

```
time.decades(2);
// 630720000000
```

```
time.centuries(5);
// 15768000000000
```

```
time.millennia(1000);
// 31536000000000000
```

```
time.fortnights(100);
// 120960000000
```
